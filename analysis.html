print(f"Total number of employees: {len(df)}")
print(f"\nDataset shape: {df.shape}")

# Calculate frequency count for "Operations" department
operations_count = df[df['department'] == 'Operations'].shape[0]
print(f"\n=== FREQUENCY COUNT ===")
print(f"Number of employees in Operations department: {operations_count}")

# Display all department counts
print("\n=== All Department Counts ===")
department_counts = df['department'].value_counts()
print(department_counts)

# Create histogram showing distribution of departments
plt.figure(figsize=(10, 6))
sns.set_style("whitegrid")

# Create the histogram
department_counts_sorted = df['department'].value_counts().sort_index()
colors = ['#21808d' if dept == 'Operations' else '#5e5240' 
          for dept in department_counts_sorted.index]

plt.bar(department_counts_sorted.index, department_counts_sorted.values, 
        color=colors, alpha=0.8, edgecolor='black', linewidth=1.2)

plt.xlabel('Department', fontsize=12, fontweight='bold')
plt.ylabel('Frequency (Number of Employees)', fontsize=12, fontweight='bold')
plt.title('Distribution of Employees Across Departments', 
          fontsize=14, fontweight='bold', pad=20)
plt.xticks(rotation=45, ha='right')
plt.grid(axis='y', alpha=0.3)

# Add value labels on top of bars
for i, (dept, count) in enumerate(department_counts_sorted.items()):
    plt.text(i, count + 0.5, str(count), ha='center', 
             fontweight='bold', fontsize=10)

plt.tight_layout()
plt.savefig('department_distribution.png', dpi=300, bbox_inches='tight')
print("\n✓ Histogram saved as 'department_distribution.png'")
